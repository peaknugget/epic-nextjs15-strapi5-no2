// pages/login.js
import React from "react";

const providers = ["google", "naver", "kakao", "facebook"];

export default function LoginPage() {
  const handleLogin = (provider) => {
    window.location.href = `/auth/${provider}`; // Strapi 인증 라우트로 리디렉션
  };

  return (
    <div className="flex flex-col items-center justify-center h-screen">
      <h1 className="text-2xl font-bold mb-6">로그인</h1>
      {providers.map((provider) => (
        <button
          key={provider}
          onClick={() => handleLogin(provider)}
          className="bg-blue-500 text-white px-4 py-2 rounded mb-3"
        >
          {provider.charAt(0).toUpperCase() + provider.slice(1)} 로그인
        </button>
      ))}
    </div>
  );
}


// pages/auth/[provider].js
import { useRouter } from "next/router";
import { useEffect } from "react";

export default function AuthProviderPage() {
  const router = useRouter();
  const { provider } = router.query;

  useEffect(() => {
    if (provider) {
      const redirectUri = encodeURIComponent(
        `${process.env.NEXT_PUBLIC_APP_URL}/auth/${provider}/callback`
      );
      const authUrl = getProviderAuthUrl(provider, redirectUri);
      window.location.href = authUrl;
    }
  }, [provider]);

  const getProviderAuthUrl = (provider, redirectUri) => {
    switch (provider) {
      case "google":
        return `https://accounts.google.com/o/oauth2/v2/auth?client_id=${process.env.NEXT_PUBLIC_GOOGLE_CLIENT_ID}&redirect_uri=${redirectUri}&response_type=code&scope=openid email profile`;
      case "naver":
        return `https://nid.naver.com/oauth2.0/authorize?client_id=${process.env.NEXT_PUBLIC_NAVER_CLIENT_ID}&response_type=code&redirect_uri=${redirectUri}&state=${process.env.NEXT_PUBLIC_NAVER_STATE}`;
      case "kakao":
        return `https://kauth.kakao.com/oauth/authorize?client_id=${process.env.NEXT_PUBLIC_KAKAO_CLIENT_ID}&redirect_uri=${redirectUri}&response_type=code`;
      case "facebook":
        return `https://www.facebook.com/v16.0/dialog/oauth?client_id=${process.env.NEXT_PUBLIC_FACEBOOK_CLIENT_ID}&redirect_uri=${redirectUri}&response_type=code&scope=email`;
      default:
        return "/";
    }
  };

  return <div>로그인 처리 중...</div>;
}
